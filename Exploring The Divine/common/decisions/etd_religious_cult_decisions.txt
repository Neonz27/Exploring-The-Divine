# Decision To Trigger The Found A Religious Cult Event
etd_found_religious_cult_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	
	title = etd_found_religious_cult_decision_title
	selection_tooltip = etd_found_religious_cult_decision_tooltip
	desc = etd_found_religious_cult_decision_desc
	confirm_text = etd_found_religious_cult_decision_confirm

	is_shown = {
		age >= etd_found_religious_cult_age_requirement
		is_imprisoned = etd_found_religious_cult_can_be_imprisoned
		is_landed = etd_found_religious_cult_must_be_landed
		is_ruler = etd_found_religious_cult_must_be_ruler
		
		NOT = {
			has_trait = cult_leader
		}
	}

	is_valid = {
		OR = {
			diplomacy >= etd_found_religious_cult_diplomacy_requirement
			intrigue >= etd_found_religious_cult_intrigue_requirement
			learning >= etd_found_religious_cult_learning_requirement
		}
	}

	cost = {
		piety = etd_found_religious_cult_piety_requirement
		prestige = etd_found_religious_cult_prestige_requirement
	}

	effect = {
		trigger_event = etd_religious_cult_events.1
	}
	
	ai_check_frequency = etd_found_religious_cult_ai_check_frequency

	ai_potential = {
		always = etd_found_religious_cult_ai_potential
	}

	ai_will_do = {
		base = 5
	}
}
